139void statisticalOutlierRemovalFilter(
140 const typename pcl::PointCloud<PointT>::Ptr& cloud_in,
141 typename pcl::PointCloud<PointT>::Ptr& cloud_out, int nb_neighbors,
142 float std_ratio, bool filter_nan) {
143 pcl::StatisticalOutlierRemoval<PointT> sor;
144 if (filter_nan) {
145 typename pcl::PointCloud<PointT>::Ptr temp_cloud(
146 new pcl::PointCloud<PointT>);
147 filterNaN<PointT>(cloud_in, temp_cloud);
148 sor.setInputCloud(temp_cloud);
149 } else {
150 sor.setInputCloud(cloud_in);

152 sor.setMeanK(nb_neighbors); // 设置计算平均距离的邻居点数
153 sor.setStddevMulThresh(std_ratio); // 设置标准差倍数阈值
154 sor.filter(*cloud_out);

190template <typename PointT>
191typename pcl::FieldComparison<PointT>::Ptr removalCondition(
192 const std::string& field_name, float val, const std::string& comparison) {
193 pcl::ComparisonOps::CompareOp op;
194 if (comparison == ">") {
195 op = pcl::ComparisonOps::GT;
196 } else if (comparison == ">=") {
197 op = pcl::ComparisonOps::GE;
198 } else if (comparison == "<") {
199 op = pcl::ComparisonOps::LT;
200 } else if (comparison == "<=") {
201 op = pcl::ComparisonOps::LE;
202 } else if (comparison == "=") {
203 op = pcl::ComparisonOps::EQ;
204 } else {
205 throw std::runtime_error("Invalid comparison operator");
206 exit(-1);
207 }
208 typename pcl::FieldComparison<PointT>::Ptr cond(
209 new pcl::FieldComparison<PointT>(field_name, op, val));
210 return cond;
211}

257template <typename PointT>
258void conditionRemovalFilter(
259 const typename pcl::PointCloud<PointT>::Ptr& cloud_in,
260 typename pcl::PointCloud<PointT>::Ptr& cloud_out,
261 const typename std::vector<typename pcl::FieldComparison<PointT>::Ptr>&
262 cond_list,
263 const std::string& cond_op, bool is_negative) {
264 typename pcl::ConditionBase<PointT>::Ptr conds;
265 if (cond_op == "and") {
266 conds.reset(new pcl::ConditionAnd<PointT>());
267 } else {
268 conds.reset(new pcl::ConditionOr<PointT>());
269 }
270 for (auto cond : cond_list) {
271 conds->addComparison(cond);
272 }
273
274 pcl::ConditionalRemoval<PointT> cond_rem(true);
275 cond_rem.setCondition(conds);
276 cond_rem.setInputCloud(cloud_in);
277 cond_rem.setKeepOrganized(true);
278 cond_rem.filter(*cloud_out);
279
280 if (is_negative) {
281 pcl::PointIndices::Ptr removed_indices(new pcl::PointIndices());
282 cond_rem.getRemovedIndices(*removed_indices);
283 cloud_out->clear();
284 extractIndices<PointT>(cloud_in, cloud_out, removed_indices, false);
285 }

341template <typename PointT>
342void cropPointCloudBy8Vertices(
343 const typename pcl::PointCloud<PointT>::Ptr& cloud_in,
344 const std::vector<Eigen::Vector3f>& box_8_vertices,
345 typename pcl::PointCloud<PointT>::Ptr& cloud_out, float epsilon = 1e-6f) {
346 // ===================== 步骤2：将8个顶点转为点云，拟合OBB核心参数
347 // =====================
348
349 typename pcl::PointCloud<PointT>::Ptr box_cloud(
350 new pcl::PointCloud<PointT>);
351 box_cloud->resize(8);
352 for (int i = 0; i < 8; ++i) {
353 PointT pt;
354 pt.x = box_8_vertices[i].x();
355 pt.y = box_8_vertices[i].y();
356 pt.z = box_8_vertices[i].z();
357 box_cloud->at(i) = pt;
358 }
359
360 OBBDate obb;
361 util::calcOBB<PointT>(box_cloud, obb);
362
363 PointT obb_center;
364 obb_center.x = obb.center.x();
365 obb_center.y = obb.center.y();
366 obb_center.z = obb.center.z();
367
368 Eigen::Matrix3f obb_rot = obb.axes; // 修正：使用axes而非rotation
369 Eigen::Vector3f obb_size =
370 obb.extents * 2.0f; // 修正：extents是半尺寸，需乘以2
371
372 // ===================== 步骤3：计算OBB局部系筛选阈值 =====================
373 // 注意：现在obb_size已经是全尺寸
374 float x_thresh = obb.extents.x() - epsilon; // 直接使用extents
375 float y_thresh = obb.extents.y() - epsilon;
376 float z_thresh = obb.extents.z() - epsilon;
377
378 // ===================== 步骤4：遍历点云，精准筛选OBB内的点
379 // =====================
380 cloud_out->clear();
381 cloud_out->reserve(cloud_in->size());
382
383 for (const auto& p : *cloud_in) {
384 // 坐标变换：世界系 → OBB局部系（平移+旋转）
385 Eigen::Vector3f p_w(p.x, p.y, p.z);
386 Eigen::Vector3f p_trans =
387 p_w - Eigen::Vector3f(obb_center.x, obb_center.y, obb_center.z);
388 Eigen::Vector3f p_local = obb_rot.transpose() * p_trans;
389
390 // 局部系中OBB轴对齐，判断点是否在盒内
391 if (std::fabs(p_local.x()) <= x_thresh &&
392 std::fabs(p_local.y()) <= y_thresh &&
393 std::fabs(p_local.z()) <= z_thresh) {
394 cloud_out->push_back(p);
395 }
396 }
397
398 // ===================== 步骤5：打印调试信息 =====================
399 std::cout << "===== OBB裁剪结果 =====" << std::endl;
400 std::cout << "OBB中心：(" << obb_center.x << "," << obb_center.y << ","
401 << obb_center.z << ")\n";
402 std::cout << "原始点云数：" << cloud_in->size() << " | 盒内点云数："
403 << cloud_out->size() << "\n";
404}
