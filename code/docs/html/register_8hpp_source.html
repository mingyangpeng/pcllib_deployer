52template <typename PointT>
53bool icpPToPoint(const typename pcl::PointCloud<PointT>::Ptr& cloud_src,
54 const typename pcl::PointCloud<PointT>::Ptr& cloud_tgt,
55 Eigen::Matrix4f& transform, int max_iter, float max_corr_dist,
56 float trans_epsilon, float fitness_epsilon) {
57 typename pcl::PointCloud<PointT>::Ptr cloud_icp(
58 new pcl::PointCloud<PointT>);
59 pcl::IterativeClosestPoint<PointT, PointT> icp;
60 icp.setInputSource(cloud_src);
61 icp.setInputTarget(cloud_tgt);
62 icp.setMaxCorrespondenceDistance(max_corr_dist);
63 icp.setMaximumIterations(max_iter);
64 icp.setTransformationEpsilon(trans_epsilon);
65 icp.setEuclideanFitnessEpsilon(fitness_epsilon);
66 icp.align(*cloud_icp);
67 if (icp.hasConverged()) {
68 transform = icp.getFinalTransformation();
69 return true;
70 } else {
71 PCL_ERROR("ICP failed to converge!");
72 return false;
73 }
74}
