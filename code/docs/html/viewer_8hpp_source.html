170// 显示pca方向
171template <typename PointT>
172void showPCA(const typename pcl::PointCloud<PointT>::Ptr& cloud,
173 const Eigen::Vector4f& centroid,
174 const Eigen::Matrix3f& eigenvectors,
175 const Eigen::Vector3f& eigenvalues) {
176 pcl::visualization::PCLVisualizer::Ptr viewer(
177 new pcl::visualization::PCLVisualizer("PCA Visualization"));
178 viewer->setBackgroundColor(0, 0, 0); // 黑色背景
179 // 添加点云（白色）
180 pcl::visualization::PointCloudColorHandlerCustom<PointT> cloud_color(
181 cloud, 255, 255, 255);
182 viewer->addPointCloud<PointT>(cloud, cloud_color, "cloud");
183 viewer->setPointCloudRenderingProperties(
184 pcl::visualization::PCL_VISUALIZER_POINT_SIZE, 2, "cloud");
185
186 // 质心坐标（x,y,z）
187 float cx = centroid(0), cy = centroid(1), cz = centroid(2);
188 // 计算箭头长度：特征值开方（归一化，避免长度过长/过短）
189 float scale = 1.0 / eigenvalues.maxCoeff(); // 归一化系数
190 Eigen::Vector3f ev1 = eigenvectors.col(0) * eigenvalues(0) * scale; // PC1
191 Eigen::Vector3f ev2 = eigenvectors.col(1) * eigenvalues(1) * scale; // PC2
192 Eigen::Vector3f ev3 = eigenvectors.col(2) * eigenvalues(2) * scale; // PC3
193
194 // 绘制PC1：红色箭头（第一主成分，最长）
195 viewer->addLine(PointT(cx, cy, cz),
196 PointT(cx + ev1(0), cy + ev1(1), cz + ev1(2)), 1, 0, 0,
197 "PC1");
198 viewer->setShapeRenderingProperties(
199 pcl::visualization::PCL_VISUALIZER_LINE_WIDTH, 3, "PC1");
200 // 绘制PC2：绿色箭头（第二主成分）
201 viewer->addLine(PointT(cx, cy, cz),
202 PointT(cx + ev2(0), cy + ev2(1), cz + ev2(2)), 0, 1, 0,
203 "PC2");
204 viewer->setShapeRenderingProperties(
205 pcl::visualization::PCL_VISUALIZER_LINE_WIDTH, 3, "PC2");
206 // 绘制PC3：蓝色箭头（第三主成分）
207 viewer->addLine(PointT(cx, cy, cz),
208 PointT(cx + ev3(0), cy + ev3(1), cz + ev3(2)), 0, 0, 1,
209 "PC3");
210 viewer->setShapeRenderingProperties(
211 pcl::visualization::PCL_VISUALIZER_LINE_WIDTH, 3, "PC3");
212 // 绘制质心：黄色球体
213 viewer->addSphere(PointT(cx, cy, cz), 0.02, 1, 1, 0, "centroid");
214}

216// 显示OBB
217template <typename PointT>
218void showOBB(const typename pcl::PointCloud<PointT>::Ptr& cloud,
219 const OBBDate& obb,
220 const std::string& window_name = "OBB Viewer") {
221 boost::shared_ptr<pcl::visualization::PCLVisualizer> viewer(
222 new pcl::visualization::PCLVisualizer(window_name));
223 viewer->setBackgroundColor(0, 0, 0);
224
225 // 添加点云
226 pcl::visualization::PointCloudColorHandlerCustom<PointT> cloud_color(
227 cloud, 175, 238, 238);
228 viewer->addPointCloud<PointT>(cloud, cloud_color, "cloud");
229 viewer->setPointCloudRenderingProperties(
230 pcl::visualization::PCL_VISUALIZER_POINT_SIZE, 3, "cloud");
231
232 // 添加OBB的8个顶点
233 pcl::PointCloud<pcl::PointXYZ>::Ptr obb_vertices(
234 new pcl::PointCloud<pcl::PointXYZ>);
235 for (const auto& vertex : obb.vertices) {
236 obb_vertices->push_back(
237 pcl::PointXYZ(vertex.x(), vertex.y(), vertex.z()));
238 }
239
240 // 绘制OBB的边
241 std::vector<std::pair<int, int>> edges = {
242 {0, 1}, {1, 2}, {2, 3}, {3, 0}, // 底面
243 {4, 5}, {5, 6}, {6, 7}, {7, 4}, // 顶面
244 {0, 4}, {1, 5}, {2, 6}, {3, 7} // 连接顶面和底面的边
245 };
246
247 // 为每条边添加线段
248 for (size_t i = 0; i < edges.size(); ++i) {
249 const auto& edge = edges[i];
250 std::string line_id = "line_" + std::to_string(i);
251 viewer->addLine<pcl::PointXYZ>(obb_vertices->points[edge.first],
252 obb_vertices->points[edge.second], 1.0,
253 0.0, 1.0, // 红色
254 line_id);
255 viewer->setShapeRenderingProperties(
256 pcl::visualization::PCL_VISUALIZER_LINE_WIDTH, 2, line_id);
257 }
258
259 // 添加坐标系
260 viewer->addCoordinateSystem(1.0);
261 viewer->initCameraParameters();
262
263 while (!viewer->wasStopped()) {
264 viewer->spinOnce();
265 }
266}
