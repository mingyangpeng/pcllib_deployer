111template <typename PointT>
112bool calcOBB(const typename pcl::PointCloud<PointT>::Ptr& cloud, OBBDate& obb) {
113 try {
114 // 使用PCL的MomentOfInertiaEstimation计算OBB
115 pcl::MomentOfInertiaEstimation<pcl::PointXYZ> moe;
116 moe.setInputCloud(cloud);
117 moe.compute();
118
119 pcl::PointXYZ min_point, max_point, position;
120 Eigen::Matrix3f rotational_matrix;
121
122 moe.getOBB(min_point, max_point, position, rotational_matrix);
123
124 // 填充OBB结构体
125 obb.center = Eigen::Vector3f(position.x, position.y, position.z);
126 obb.min_point = Eigen::Vector3f(min_point.x, min_point.y, min_point.z);
127 obb.max_point = Eigen::Vector3f(max_point.x, max_point.y, max_point.z);
128 obb.axes = rotational_matrix;
129
130 // 计算半尺寸
131 obb.extents = (obb.max_point - obb.min_point) / 2.0f;
132
133 // 计算OBB的8个顶点
134 std::vector<Eigen::Vector3f> vertices;
135 getOBBCorners(obb, vertices);
136 obb.vertices = vertices;
137
138 // 计算原始坐标系下的最小和最大点
139 obb.min_point = vertices[0];
140 obb.max_point = vertices[0];
141 for (size_t i = 1; i < vertices.size(); ++i) {
142 obb.min_point = obb.min_point.cwiseMin(vertices[i]);
143 obb.max_point = obb.max_point.cwiseMax(vertices[i]);
144 }
145
146 return true;
147 } catch (const std::exception& e) {
148 std::cerr << "Error computing OBB: " << e.what() << std::endl;
149 return false;
150 }
151}
